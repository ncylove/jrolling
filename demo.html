jQuery Rolling

기본 옵션

default = {
	'items': null, // 대상
	'width': null, // 대상 width
	'height': null, // 대상 height
	'frame': 500, // 흐르는 속도
	'delay': 2000, // 초 후에 흐름
	'move': 'left', // 흐름 방향
	'auto': true, // 자동 흐름 설정
	'prev': null, // 이전 클릭 이벤트
	'next': null // 다음 클릭 이벤트
}

기본 흐름

	var jrolling = $('#jrolling').jrolling({
		'items': '#jrolling ul',
		'width': '220px',
		'height': '84px',
		'move': 'up',
		'auto': false,
		'prev': '#prev',
		'next': '#next'
	});

	jrolling.target.on('motionAfter', function(event, index) {
		$('#jrolling_text').text(index);
	});

임의적인 배열 데이터의 흐름

	var jrolling2 = $('#jrolling2').jrolling({
		'items': [ 'a', 'b', 'c' ],
		'width': '220px',
		'height': '220px',
		'auto': true,
		'prev': '#prev2',
		'next': '#next2'
	});

	jrolling2.target.on('motionBefore', function(event, index) {
		$('#jrolling2_text').text(index);
	});

이벤트 트리거

motionBefore : 이동 전
motionAfter : 이동 후
